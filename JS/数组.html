                                      1.求一组数中的所有数的和和平均值
                                    var arr = [11,22,33];

                                    2.求一组数中的最大值和最小值，以及所在位置

                                    求最大值：两两比较；我设置一个值，然后和数组中的每一个比较；
                                            如果我的大，那么就继续比较，如果数组中的元素大，那么赋值给定义的变量;继续比较;
                                    var arr = [-1,-2,-31,-4,-5];
                                    //bug:不能用数组之外的进行比较，万一比数组中的每一个都大，就不对了；
                                    var max = arr[0];//第一个和最后一个最好，因为数组可以少比较一次；

                                    //for循环遍历出每一个数组中的元素，进行比较
                                    for(var i=1;i<arr.length;i++){//能少比较一次算一次;
                                        //判断：如果数组中的元素大，给max重新赋值；否则不用管;继续比较下一个;
                                        if(max < arr[i]){
                                            //数组中的元素如果大，就重新赋值给max；
                                            max = arr[i];
                                        }
                                        //     else{
                                         //继续;
                                        //  }
                                    }
                                    alert(max);

                                    最大值和最小值,及所在位置;
                                    var arr = [1,2,31,4,5];
                                    var maxValue = arr[0];//定义数组中的第一个;
                                    var minValue = arr[0];//定义数组中的第一个;
                                    //位置(上面设置的哪一项，一定要保证索引值也对应)
                                    var maxIndex = 0;
                                    var minIndex = 0;

                                    //遍历每一项
                                    for(var i=1;i<arr.length;i++){
                                        //比较谁大SEI给max/minValue重新赋值;
                                        if(maxValue < arr[i]){
                                            maxValue = arr[i];//数组中的元素如果大于最大值，就为他重新赋值;
                                            //赋值索引值
                                            maxIndex = i;
                                        }
                                        //大于最大值，小于最小值才重新赋值;而不是不大于最大值就赋值(不一定时最小值);
                                        if(minValue > arr[i]){
                                            minValue = arr[i];
                                            //赋值索引值
                                            minIndex = i;
                                        }
                                    }

数组反转 

                                     反转数组：让元素前后交换位置;
                                        1.使用新数组接收;(不修改原数组)
                                            a.正向遍历,反向添加;
                                            b.反向遍历,正向添加;
                                        2.不使用新数组接收;(修改原数组)
                                            a.原数组中的元素前后交换位置;(第一和倒数第一交换，第二和倒数第二交换...)

                                    var arr = [1,2,3,4,5,6];
                                    1.使用新数组接收;(不修改原数组)
                                    var newArr = [];
                                        a.正向遍历,反向添加;
                                    for(var i=0;i<arr.length;i++){
                                        //反向添加;
                                        newArr[newArr.length] = arr[arr.length-1-i];//第一次添加的是倒数第一个;
                                            //倒数第一个： arr.length-1-0;//i代表的就是要减去的数值;
                                            //倒数第二个： arr.length-1-1;//i代表的就是要减去的数值;
                                            //倒数第三个： arr.length-1-2;//i代表的就是要减去的数值;
                                            //...
                                    }

                                        b.反向遍历,正向添加;
                                    for(var i=arr.length-1;i>=0;i--){//反向遍历，索引值从最大到最小
                                        //添加元素；怎么遍历怎么添加;
                                        newArr[newArr.length] = arr[i];
                                    }

                                    2.不使用新数组接收;(修改原数组)
                                        a.原数组中的元素前后交换位置;(第一和倒数第一交换，第二和倒数第二交换...)
                                            bug: 6个元素只要3次，5个元素需要2-3；
                                    for(var i=0;i<arr.length/2;i++){//不能写等号，偶数就会多一次；
                                        //前后交换位置;(第一和倒数第一交换，第二和倒数第二交换...)
                                        var temp = arr[i];
                                        arr[i] = arr[arr.length-1-i];//第一次是倒数第一个，第二次是倒数第二个...
                                        arr[arr.length-1-i] = temp;
                                    }

                                    console.log(arr);
                                    console.log(newArr);

冒泡排序 

                                   冒泡排序：给元素排列顺序;
                                   思路：共比较元素个数-1轮，每轮比较元素个数-1次(轮数和次数都用for循环)
                                   满足某一条件，相邻两个元素就前后交换位置；
                                   如果想要优化冒泡排序：只能从两个方向：1.次数;    2.轮数;
                                   1.次数;(每多比较一轮少比较一次)(每轮找到一个最合适的)
                                   2.轮数;(如果这一轮比较中已经完成就不需要下一轮在比较了)
                                            在整轮比较过程中，没有任何两个相邻的元素前后交换位置;(跳出外循环)
                                    var arr = [1,2,3,4,5,6,7];
                                    console.log(arr);
                                    var n = 0;//次数
                                    var m = 0;//轮数
                                    //外循环控制轮数，内循环控制次数，轮数和次数都是元素个数-1;
                                    for(var i=0;i<arr.length-1;i++){
                                        //开闭原则：类似一个开关，默认是开启的，有一个说关闭，我们就把他关闭掉;
                                        var bool = true;//必须写在两个for循环之间,下一轮开始的时候才会初始化bool值为true；
                                        //1.次数;(每多比较一轮少比较一次)(每轮找到一个最合适的)
                                            //第一轮-0；
                                            //第二轮-1；
                                            //第三轮-2；......
                                        for(var j=0;j<arr.length-1-i;j++){//每多比较一轮少比较一次
                                            //满足条件,交换位置
                                            if(arr[j] > arr[j+1]){
                                                var temp = arr[j];
                                                arr[j] = arr[j+1];
                                                arr[j+1] = temp;
                                                //能够尽量就说明交换位置了
                                                bool = false;//false将不允许跳出外循环;
                                            }
                                            n++;
                                        }
                                        m++; //如果在m++之前，未来会少一次;
                                        //判断bool值，如果是true，就没有任何元素前后交换位置，
                                                    //如果是false，就说明有元素交换位置，还要继续比较;
                                        if(bool){
                                            break;
                                        }
                                    }
                                    console.log(arr);console.log(n);console.log(m);

                                                        array[j+1] = temp;
                                                    }
                                                }
                                            }
                                            //返回数组
